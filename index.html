試す場合はtinkercadなどのシュミレーションできるwebサイトか，

問題1

LEDを1秒ごとに点滅させてください

void setup()
{
pinMode(4, OUTPUT);
}
void loop()
{
digitalWrite(4, HIGH);
delay(1000);
digitalWrite(4, LOW);
delay(1000);
}


問題2

3つのLEDを1秒ごとに被らないように点滅させて下さい
例：赤1秒→青1秒→緑1秒

void setup()
{
pinMode(3, OUTPUT);
pinMode(6, OUTPUT);
pinMode(9, OUTPUT);
}
void loop()
{
digitalWrite(3, HIGH);
delay(1000); // Wait for 1000 millisecond(s)
digitalWrite(3, LOW);
digitalWrite(6, HIGH);
delay(1000);
digitalWrite(6, LOW);
digitalWrite(9, HIGH);
delay(1000);
digitalWrite(9, LOW);
}


問題3

ボタンを押したらLEDが光るようにして下さい

ボタン部分の回路の意味を知りたかったらプルダウン抵抗で検索してみてください

void setup()
{
pinMode(3, OUTPUT);
pinMode(8, INPUT);
}
void loop()
{
int flag=digitalRead(8);
if(flag==1){
digitalWrite(3,HIGH);
}
else{
digitalWrite(3,LOW);
}
}

問題5

シリアル通信を使ってAが入力されたら光ってBが入力されたら消えるようにしてください

void setup()
{
pinMode(3, OUTPUT);
Serial.begin( 9600 );
}
void loop()
{
char moji='C';
if(Serial.available()>0){
moji=Serial.read();
}
if(moji=='A'){
digitalWrite(3,HIGH);
}
else if(moji=='B'){
digitalWrite(3,LOW);
}
}


問題6

LEDの光をだんだん明るくなるようにしてください

void setup()
// LEDの明るさを段階的に変更するための変数
int brightness = 0;
// 明るさを増減させる量
const int increment = 10;
void setup() {
// LEDピンを出力モードに設定
pinMode(ledPin, OUTPUT);
} void loop() {
// LEDの明るさを設定
analogWrite(ledPin, brightness);
// 明るさをインクリメント
brightness += increment;
// brightnessが0から255の間にあることを確認
if (brightness > 255) {
brightness = 0; // brightnessを0にリセット
}
// 1秒（1000ミリ秒）待機
delay(1000);
}
